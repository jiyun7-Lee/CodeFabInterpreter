# 1. 정상동작 테스트 (기초 기능 테스트)

**1) 표현식 / 연산자 / 우선순위 / 진리값**

```
// --- 산술 , 연산자 우선순위 (곱셈/나눗셈이 덧셈/뺄셈보다 먼저) ---
print 1 + 2 * 3;        // expect: 7
print (1 + 2) * 3;      // expect: 9
print 10 - 4 - 3;       // expect: 3   
print 8 / 2 / 2;        // expect: 2   
print -3 + 2;           // expect: -1  

// --- 비교 , 동등성 --- 
print 1 < 2;            // expect: true
print 3 > 5;            // expect: false

// --- 문자열 연결 (+) ---
print "Hello, " + "CodeFab!";   // expect: Hello, CodeFab!

// --- 숫자 출력 포맷 (정수는 .0 없이 출력되어야 함) ---
print 5;                // expect: 5
print 5.0;              // expect: 5
print 3.14;             // expect: 3.14

// --- boolean 리터럴 출력 ---
print true;             // expect: true
print false;            // expect: false
```



**2) 변수, 할당, 블록 스코프, 변수 shawdowing**

```
// --- 선언 & 사용 ---
var a = 10;
var b = 20;
print a + b;            // expect: 30


// --- 재할당 --- 
a = a + 5;
print a;                // expect: 15

// --- 블록 스코프 & 변수 shawdowing ---
// 해당 기능은 Prompt 모드에서 테스트보다는 File 입력모드에서 진행 권장드립니다. (멀티라인입력 문제 해결시 가능) 
// 이후에 File 입력모드가 추가되는데 해당 모드에서 테스트해주세요.
var x = "global";
{
  var x = "inner";
  print x;              // expect: inner
}
print x;                // expect: global  (블록 밖은 영향 없음)


// --- 안쪽 블록에서 바깥 변수 수정은 가능 ---
// 해당 기능은 Prompt 모드에서 테스트보다는 File 입력모드에서 진행 권장드립니다. (멀티라인입력 문제 해결시 가능) 
// 이후에 File 입력모드가 추가되는데 해당 모드에서 테스트해주세요.
var count = 0;
{
  count = count + 1;    // 같은 이름 재선언이 아니라 바깥 변수 수정
}
print count;            // expect: 1


// 중첩 스코프 해석
// 해당 기능은 Prompt 모드에서 테스트보다는 File 입력모드에서 진행 권장드립니다. (멀티라인입력 문제 해결시 가능) 
// 이후에 File 입력모드가 추가되는데 해당 모드에서 테스트해주세요.
var outer = "A";
{
  var inner = "B";
  {
    print outer + inner;  // expect: AB
  }
}

```



**3) 제어 흐름 (if/else, for)**
```
// --- if / else ---
if (true) print "bbq";    // expect: bbq

if (false) print "no"; else print "kfc";  // expect: kfc


// --- else 는 가장 가까운 if 에 결합 ---
// 해당 기능은 Prompt 모드에서 테스트보다는 File 입력모드에서 진행 권장드립니다.(멀티라인입력 문제 해결시 가능) 
// 이후에 File 입력모드가 추가되는데 해당 모드에서 테스트해주세요.
if (true)
  if (false) print "kfc";
  else print "bbq";        // expect: bbq



// --- for 반복문 ---
for (var j = 0; j < 3; j = j + 1) { print j; }// expect: 0, 1, 2


```

# 2. 에러 검출 테스트 (각 스크립트는 "에러를 내야" 정상)
오류 메시지는 팀에서 정한 메시지를 출력.


**1) 구문 에러**
```
// 세미콜론 누락
// 기대: "Expect ';' after value." 류의 메시지
print 1 + 2
```

```
// 닫는 괄호 누락
// 기대: "Expect ')' after expression." 류의 메시지
print (1 + 2;
```

```
// 잘못된 할당 대상
// 기대: "Invalid assignment target." 류의 메시지
var a = 1;
var b = 2;
a + b = 3;
```

```
// 표현식이 와야 할 자리에 엉뚱한 토큰
// 기대: "Expect expression." 류의 메시지
print * 5;
```



**2) Checker Unit 에서 검출하는 에러 (정적에러)**
```
// 지역 변수를 자기 초기화식에서 읽기 (선언전 사용 오류)
// 기대: "Can't read local variable in initializer." 류의 메시지
{ var a = a;}
```

```
// 같은 로컬 스코프 중복 선언 오류 
// 기대: "Already a variable with this name in this scope." 류의 메시지
{ var a = "hi"; var a = 3; }
```



**3) 실행중 발생하는 런타임 에러**
```
// 정의되지 않은 변수 참조
// 기대: "Undefined variable 'notDefined'." (+ 줄 번호) 류의 메시지 
print notDefined;
```

```
// + 연산자에 숫자/문자열 혼용
// 기대: "Operands must be two numbers or two strings." 류의 메시지
print 1 + "HI";
```

```
// 숫자가 아닌 값에 단항 마이너스
// 기대: "Operand must be a number." 류의 메시지
print -"FabCoding";
```
